Question 6

// Input

let x1 = +prompt("Enter Coordinate of X1");

let y1 = +prompt("Enter Coordinate of Y1");

let x2 = +prompt("Enter Coordinate of X2");

let y2 = +prompt("Enter Coordinate of Y2");

// Process

let dstnc = Math.sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));

// Output

alert(`Distance between the two points is ${dstnc}`);
